💻 参考代码

// P2895 Meteor Shower S — BFS + 时间预处理 #include <bits/stdc++.h> using namespace std; const int N = 605, INF = 0x3f3f3f3f; int des[N][N]; // 每个格子最早被烧焦的时间 bool vis[N][N]; int dx[] = {0,0,1,-1}; int dy[] = {1,-1,0,0}; struct Node { int x, y, t; }; int bfs() { queue<Node> q; q.push({0,0,0}); vis[0][0] = true; while (!q.empty()) { auto [x, y, t] = q.front(); q.pop(); if (des[x][y] == INF) return t; // 到达安全格子！ for (int d = 0; d < 4; d++) { int nx = x + dx[d], ny = y + dy[d]; if (nx < 0 || ny < 0) continue; if (vis[nx][ny]) continue; if (t + 1 < des[nx][ny]) { vis[nx][ny] = true; q.push({nx, ny, t + 1}); } } } return -1; } int main() { int M; cin >> M; memset(des, 0x3f, sizeof(des)); for (int i = 0; i < M; i++) { int x, y, t; cin >> x >> y >> t; des[x][y] = min(des[x][y], t); for (int d = 0; d < 4; d++) { int nx = x + dx[d], ny = y + dy[d]; if (nx >= 0 && ny >= 0) des[nx][ny] = min(des[nx][ny], t); } } cout << bfs() << endl; }

🧠 解题思路

① 预处理时间数组：des[x][y] 记录格子 (x,y) 最早被烧焦的时间，初始 INF（安全）。
② 读入每颗流星：更新落点 + 四周 4 格的 des 值（取 min，因为可能被多颗流星覆盖）。
③ BFS：从 (0,0, t=0) 开始，扩展时判断 t+1 < des[nx][ny] 才能走。
④ 终止条件：遇到 des[x][y] == INF 的格子 → 安全！返回当前时间 t。
⑤ 易错点：流星 ≤ 300，但贝茜可走到 >300；数组要开大（≥605）。

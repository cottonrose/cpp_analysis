📋 动态规划五步法（排序 DP）
① 划分阶段
按高度升序处理每个格子。
将所有格子按高度从小到大排序，保证处理当前格时，更低的相邻格子 f 值已算好。
② 状态表示
f[i][j] = 从 (i,j) 出发能获得的最长滑雪路径长度（包含自身）
③ 状态转移方程
遍历上、下、左、右四个方向：
若 a[nx][ny] < a[x][y] 且 (nx,ny) 在界内：
  f[x][y] = max(f[x][y], f[nx][ny])
处理完所有更低邻居后 +1（计入自身）：
f[x][y]++
④ 边界条件
f 数组初始化为 0
若四周都更高（局部最低点）：
f = 0 → ++ → 1（只能待在原地）
⑤ 答案
ans = max(所有 f[i][j])
即所有可能起点中，能滑出的最长路径长度。
💻 DP 排序版参考代码 
// P1434 [SHOI2002] 滑雪 — DP 排序版  O(nm log nm)
#include <bits/stdc++.h>
using namespace std;

const int N = 105;
int n, m, a[N][N], f[N][N], tot, ans;
int dx[4] = {0,0,1,-1};
int dy[4] = {1,-1,0,0};

struct node {
    int h, x, y;             // 高度，横坐标，纵坐标
    bool operator<(const node &x)const {
        return h < x.h;          // 按高度升序
    }
} b[N * N];

int main() {
    cin >> n >> m;
    for (int i = 1; i <= n; ++i)
        for (int j = 1; j <= m; ++j) {
            cin >> a[i][j];
            b[++tot] = {a[i][j], i, j};     // ① 收集所有格子
        }

    sort(b + 1, b + 1 + tot);           // ① 按高度升序排序

    for (int i = 1; i <= tot; ++i) {     // ① 按排序后顺序处理
        int x = b[i].x, y = b[i].y;
        for (int j = 0; j <= 3; ++j) {    // ③ 检查四个方向
            int _x = x + dx[j], _y = y + dy[j];
            if (a[_x][_y] < a[x][y])     // 找到更低邻居
                f[x][y] = max(f[x][y], f[_x][_y]);
        }
        f[x][y]++;                         // ③ 计入自身
        ans = max(ans, f[x][y]);           // ⑤ 更新答案
    }

    cout << ans;
    return 0;
}
💪 记忆化搜索优化版参考代码 
// P1434 [SHOI2002] 滑雪 — 记忆化搜索  O(nm)
/* 优化思路：
 *   DP 排序版需要排序 O(nm log nm)，且必须显式处理所有格子。
 *   记忆化搜索只需 O(nm)，只计算能到达的格子，代码更简洁。
 *   核心：f[x][y] != -1 表示已算过，直接返回，避免重复递归。
 */
#include <bits/stdc++.h>
using namespace std;

const int N = 105;
int n, m, a[N][N], f[N][N], ans;
int dx[4] = {0,0,1,-1};
int dy[4] = {1,-1,0,0};

int dfs(int x, int y) {
    if (f[x][y] != -1)             // ③ 已算过，直接返回
        return f[x][y];

    f[x][y] = 0;                     // ④ 初始为 0
    for (int i = 0; i < 4; ++i) {
        int _x = x + dx[i];
        int _y = y + dy[i];
        if (_x >= 1 && _x <= n && _y >= 1 && _y <= m
            && a[_x][_y] < a[x][y])
            f[x][y] = max(f[x][y], dfs(_x, _y));  // ③ 递归 + 取最大值
    }
    return ++f[x][y];                 // ③ 计入自身后返回
}

int main() {
    cin >> n >> m;
    for (int i = 1; i <= n; ++i)
        for (int j = 1; j <= m; ++j)
            cin >> a[i][j];

    memset(f, -1, sizeof(f));       // ④ -1 表示未计算

    for (int i = 1; i <= n; ++i)
        for (int j = 1; j <= m; ++j)
            ans = max(ans, dfs(i, j));           // ⑤ 取最大值

    cout << ans;
    return 0;
}

📋 动态规划五步法（排序 DP）
① 划分阶段
按高度升序处理每个格子。
排序保证处理当前格时，更低的相邻格子 f 值已算好。
② 状态表示
f[i][j] = 从 (i,j) 出发的最长滑雪路径长度（含自身）
③ 状态转移
遍历上下左右四个方向：
a[nx][ny] < a[x][y] → f[x][y] = max(f[x][y], f[nx][ny])
遍历完后 f[x][y]++（计入自身）
同时记录 nxt[x][y] 指向最优下一格，用于还原路径
④ 边界条件
f 初始 0。局部最低点：f = 1（只能原地）
⑤ 答案
ans = max(f[i][j])，即所有起点中能滑出的最长路径长度。
样例答案：25（经过全部 25 个格子）
💻 DP 排序版参考代码 
// P1434 [SHOI2002] 滑雪 — DP 排序版  O(nm log nm)
#include <bits/stdc++.h>
using namespace std;

const int N = 105;
int n, m, a[N][N], f[N][N], tot, ans;
int dx[4] = {0,0,1,-1};
int dy[4] = {1,-1,0,0};

struct node {
    int h, x, y;
    bool operator<(const node &x)const { return h < x.h; }
} b[N * N];

int main() {
    cin >> n >> m;
    for (int i = 1; i <= n; ++i)
        for (int j = 1; j <= m; ++j) {
            cin >> a[i][j];
            b[++tot] = {a[i][j], i, j};
        }

    sort(b + 1, b + 1 + tot);          // ① 按高度升序

    for (int i = 1; i <= tot; ++i) {
        int x = b[i].x, y = b[i].y;
        for (int j = 0; j <= 3; ++j) {       // ③ 检查四方向
            int _x = x + dx[j], _y = y + dy[j];
            if (a[_x][_y] < a[x][y])
                f[x][y] = max(f[x][y], f[_x][_y]);
        }
        f[x][y]++;                         // ③ +1
        ans = max(ans, f[x][y]);           // ⑤ 更新答案
    }

    cout << ans;
    return 0;
}
💪 记忆化搜索优化版 
// P1434 [SHOI2002] 滑雪 — 记忆化搜索  O(nm)
/* 优化：排序版需 O(nm log nm) 排序 + 显式处理全部格子。
 * 记忆化搜索只需 O(nm)，只计算能到达的格子，代码更简洁。
 */
#include <bits/stdc++.h>
using namespace std;

const int N = 105;
int n, m, a[N][N], f[N][N], ans;
int dx[4] = {0,0,1,-1};
int dy[4] = {1,-1,0,0};

int dfs(int x, int y) {
    if (f[x][y] != -1) return f[x][y];
    f[x][y] = 0;
    for (int i = 0; i < 4; ++i) {
        int _x = x + dx[i], _y = y + dy[i];
        if (_x >= 1 && _x <= n && _y >= 1 && _y <= m
            && a[_x][_y] < a[x][y])
            f[x][y] = max(f[x][y], dfs(_x, _y));
    }
    return ++f[x][y];
}

int main() {
    cin >> n >> m;
    for (int i = 1; i <= n; ++i)
        for (int j = 1; j <= m; ++j)
            cin >> a[i][j];
    memset(f, -1, sizeof(f));
    for (int i = 1; i <= n; ++i)
        for (int j = 1; j <= m; ++j)
            ans = max(ans, dfs(i, j));
    cout << ans;
    return 0;
}

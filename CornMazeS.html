// P1825 Corn Maze S — BFS + 传送门预处理 #include <bits/stdc++.h> using namespace std; struct pos { int x, y, step; }; struct spt { int x1,y1, x2,y2; }; // 传送门两端 int n, m; char a[305][305]; bool v[305][305]; int dx[]={0,1,0,-1}, dy[]={1,0,-1,0}; spt spts[26]; // 传送门 queue<pos> q; bool check(int x,int y){ if(x<0||x>=n||y<0||y>=m||a[x][y]=='#'||v[x][y]) return false; return true; } void bfs(pos s, pos f){ v[s.x][s.y]=1; q.push({s.x,s.y,0}); while(!q.empty()){ auto [x,y,step]=q.front(); q.pop(); if(x==f.x&&y==f.y){ cout<<step<<endl;return; } for(int i=0;i<4;i++){ int nx=x+dx[i], ny=y+dy[i]; if(!check(nx,ny)) continue; if(a[nx][ny]>='A'&&a[nx][ny]<='Z'){ // 传送门：跳到另一端，步数+1 int t=a[nx][ny]-'A'; int tx=nx==spts[t].x1?spts[t].x2:spts[t].x1; int ty=ny==spts[t].y1?spts[t].y2:spts[t].y1; if(v[tx][ty]) continue; v[tx][ty]=1; q.push({tx,ty,step+1}); }else{ v[nx][ny]=1; q.push({nx,ny,step+1}); } } } } int main(){ cin>>n>>m; pos s,f; for(int i=0;i<n;i++) for(int j=0;j<m;j++){ cin>>a[i][j]; if(a[i][j]=='@') s={i,j,0}; if(a[i][j]=='=') f={i,j,0}; if(a[i][j]>='A'&&a[i][j]<='Z'){ int t=a[i][j]-'A'; if(spts[t].x1==-1) spts[t].x1=i,spts[t].y1=j; else spts[t].x2=i,spts[t].y2=j; } } bfs(s,f); }

🧠 解题思路

① 预处理传送门：遍历地图，把每个字母 A~Z 的两个坐标存入 spts[26]。
② BFS 队列：记录 (x, y, step)，step 是到达该格的最少步数。
③ 四方向扩展：每步向四个方向扩展（上下左右）。
④ 传送门处理：如果下一个格子是字母，走到后立刻传送到另一端（仍是 step+1，不再额外计步）。
⑤ 访问标记：草地/终点入队时标记 visited；传送门的目标点也标记 visited，避免重复入队。
⑥ 终止：队首坐标等于终点 → 输出 step。
